# Առանձնացնել թվային արժեքներ պարունակող սյուները
data_numeric <- data[sapply(data, is.numeric)]

# Նորմավորել տվյալները
data_scaled <- scale(data_numeric)
# Հաշվել կովարիացիոն մատրիցը և վիզուալիզացնել
cov_matrix <- cov(data_scaled)
ggcorrplot(cov_matrix, type = "upper", lab = TRUE, lab_size = 3, tl.cex = 10) 

# Օգտագործել pca ֆունկցիան
pca <- princomp(data_scaled)

# Տեսնել, թե յուրաքանչյուր բաղադրիչը (բաղադրիչների քանակը = սյուների քանակ) ամբողջ տվյալների որ մասն է պարունակում

# Այս ֆունկցիան ցույց է տալիս սեփական արժեքները (որոնք համապատասխանում են հիմնական բաղադրիչների ցրվածքի համամասնությանը)՝ ամենացածրից ամենաբարձր սկզբունքով

# Այս ֆունկցիան ցույց է տալիս յուրաքանչյուր փոփոխականի ազդեցությունը առաջին երկու հիմնական բաղադրիչների վրա

fviz_pca_var(pca, col.var = "purple")

# Այս ֆունկցիան օգտագործվում է՝ ցույց տալու, թե յուրաքանչյուր հիմնական բաղադրիչը ամեն փոփոխականի քանի տոկոսն է ներկայացնում
fviz_cos2(pca, choice = "var", axes = 1:4)

# Այս ֆունկցիան միավորում է նախորդ երկու պլոտերը՝ մոտ cos^2 ունեցող փոփոխականները գունավորելով նմանատիպ գույներով
fviz_pca_var(pca, col.var = "cos2",
            gradient.cols = c("orange", "red", "blue"),
            repel = TRUE)
